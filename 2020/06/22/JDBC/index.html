执行SQL语句共有三种方式:

Sataement:执行静态SQL语句并返回结果对象

PreparedStatement:SQL语句被预编译存储在该对象中

CallableStatement:用于执行存储过程

下面代码演示了如何连接并使用预编译SQL语句:

import org.junit.Test;

import java.io.IOException;
import java.io.InputStream;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;
import java.text.ParseException;
import java.text.SimpleDateFormat;
import java.util.Date;
import java.util.Properties;

public class PreParedStatementTest {
    @Test
    public void insertTest() throws ClassNotFoundException, SQLException, IOException, ParseException {
        Connection connection = null;
        PreparedStatement ps = null;//预编译SQL语句,返回preparedstatement的实例
        try {
            InputStream is = ConnectionTest.class.getClassLoader().getResourceAsStream("jdbc.properties");
            Properties pros = new Properties();
            pros.load(is);

            String user = pros.getProperty("user");
            String password = pros.getProperty("password");
            String url = pros.getProperty("url");
            String driverClass = pros.getProperty("driverClass");//这里改用新版的com.mysql.cj.jdbc.Driver
            //加载驱动
            Class.forName(driverClass);
            //获取连接
            connection = DriverManager.getConnection(url, user, password);
            //预编译SQL语句
            String sql = "insert into test.customers(name,email,birth) values(?,?,?)";
            ps = connection.prepareStatement(sql);
            //占位符填充
            ps.setString(1, "哪吒");
            ps.setString(2, "yl@gmail.com");
            SimpleDateFormat sdf = new SimpleDateFormat("yyyy-MM-dd");
            Date date = sdf.parse("1000-1-1");
            ps.setDate(3, new java.sql.Date(date.getTime()));

            //执行,如果是查询操作有返回值则是true,如果是增删改操作则是false
            ps.execute();
        } catch (IOException | ClassNotFoundException | SQLException | ParseException e) {
            e.printStackTrace();
        } finally {
            //资源关闭
            if (ps != null) {
                ps.close();
            }
            if (connection != null) {
                connection.close();
            }
        }

    }

    @Test
    public void updateTest() {
        Connection connection = null;
        PreparedStatement ps = null;
        try {
            //获取连接
            connection = JDBCUtils.getConnection();
            //预编译SQL语句
            String sql = "update test.customers set name = ? where  id= ? ";
            ps = connection.prepareStatement(sql);
            //填充占位符
            ps.setString(1, "哪吒");
            ps.setInt(2, 19);
            //执行
            ps.execute();
        } catch (IOException | ClassNotFoundException | SQLException e) {
            e.printStackTrace();
        } finally {
            //资源关闭
            JDBCUtils.closeResource(connection, ps);
        }
    }

    public int update(String sql, Object... args) {
        Connection connection = null;
        PreparedStatement ps = null;
        try {
            connection = JDBCUtils.getConnection();
            ps = connection.prepareStatement(sql);
            for (int i = 0; i < args.length; i++) {
                ps.setObject(i + 1, args[i]);
            }
            return ps.executeUpdate();//返回DML所影响的行数,或者0表示没有任何返回(例如insert操作,返回结果为0,那么表示插入失败)
        } catch (IOException | ClassNotFoundException | SQLException e) {
            e.printStackTrace();
        } finally {
            JDBCUtils.closeResource(connection, ps);
        }
        return 0;
    }
}

最后使用可变参数实现增删改的通用写法,其中IDBCUtils工具类的实现如下:

import java.io.IOException;
import java.io.InputStream;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.PreparedStatement;
import java.sql.SQLException;
import java.util.Properties;

public class JDBCUtils {
    public static Connection getConnection() throws IOException, ClassNotFoundException, SQLException {
        InputStream is = ClassLoader.getSystemClassLoader().getResourceAsStream("jdbc.properties");
        Properties pros = new Properties();
        pros.load(is);

        String user = pros.getProperty("user");
        String password = pros.getProperty("password");
        String url = pros.getProperty("url");
        String driverClass = pros.getProperty("driverClass");//这里改用新版的com.mysql.cj.jdbc.Driver

        Class.forName(driverClass);
        return DriverManager.getConnection(url, user, password);
    }

    public static void closeResource(Connection connection, PreparedStatement ps) {
        try {
            if (ps != null) {
                ps.close();
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        try {
            if (connection != null) {
                connection.close();
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
    }
}

针对一个表的通用操作:

public Customer query(String sql, Object... args) {
    Connection connection = null;
    PreparedStatement ps = null;
    ResultSet resultSet = null;
    try {
        connection = JDBCUtils.getConnection();
        ps = connection.prepareStatement(sql);
        for (int i = 0; i < args.length; i++) {
            ps.setObject(i + 1, args[i]);
        }

        resultSet = ps.executeQuery();
        ResultSetMetaData metaData = resultSet.getMetaData();//获取结果集的元数据
        int columnCount = metaData.getColumnCount();
        if (resultSet.next()) {
            Customer customer = new Customer();
            //处理结果集中一行数据的每一列
            for (int i = 0; i < columnCount; i++) {
                Object columnVal = resultSet.getObject(i + 1);
                //获取列名
                String columnName = metaData.getColumnName(i + 1);
                //给指定customer对象的对应属性赋值,通过反射,由列明找属性
                Field field = Customer.class.getDeclaredField(columnName);
                field.setAccessible(true);
                field.set(customer, columnVal);
            }
            return customer;
        }
    } catch (IOException | ClassNotFoundException | SQLException | NoSuchFieldException | IllegalAccessException e) {
        e.printStackTrace();
    } finally {
        JDBCUtils.closeResource(connection, ps, resultSet);
    }
    return null;
}

@Test
public void testQuery(){
    String sql = "select id,name,birth,email from test.customers where id=?";
    Customer customer = query(sql, 1);
    System.out.println(customer);
}

上面代码中将结果写入一个customer对象,先用结果集元数据确定每一行有多少列,然后挨个处理,根据结果集元数据拿到了列名,然后用反射根据这个列名获取对应属性并赋值.但是当bean对象的属性名和列名不一致时就要为列起别名:在SQL语句中的每个字段后加别名.此外从元数据获取的也不能是列名了,而是别名,即getColumnLabel()方法.若是没有别名,该方法则会获取列名,故不管哪种情况都推荐使用getColumnLabel()方法.

下面演示了针对不同表的通用写法,主要是使用了泛型和反射:

public <T> T getInstance(Class<T> clazz, String sql, Object... args) {
   Connection connection = null;
   PreparedStatement ps = null;
   ResultSet resultSet = null;
   try {
       connection = JDBCUtils.getConnection();
       ps = connection.prepareStatement(sql);
       for (int i = 0; i < args.length; i++) {
           ps.setObject(i + 1, args[i]);
       }

       resultSet = ps.executeQuery();
       ResultSetMetaData metaData = resultSet.getMetaData();//获取结果集的元数据
       int columnCount = metaData.getColumnCount();
       if (resultSet.next()) {
           T t = clazz.getDeclaredConstructor().newInstance();
           //处理结果集中一行数据的每一列
           for (int i = 0; i < columnCount; i++) {
               Object columnVal = resultSet.getObject(i + 1);
               //获取列名
               String columnName = metaData.getColumnLabel(i + 1);
               //给指定customer对象的对应属性赋值,通过反射,由列明找属性
               Field field = clazz.getDeclaredField(columnName);
               field.setAccessible(true);
               field.set(t, columnVal);
           }
           return t;
       }
   } catch (IOException | ClassNotFoundException | SQLException | NoSuchFieldException | IllegalAccessException | NoSuchMethodException | InstantiationException | InvocationTargetException e) {
       e.printStackTrace();
   } finally {
       JDBCUtils.closeResource(connection, ps, resultSet);
   }
   return null;
}

            ps = connection.prepareStatement(sql);

关闭自动提交还用于MySQL的事务:

public void updateWithTransaction(Connection connection, String sql, Object @NotNull ... args) {
    //将连接作为参数传进来,是的多个DML语句共用一个连接
    PreparedStatement ps = null;
    try {
        ps = connection.prepareStatement(sql);
        for (int i = 0; i < args.length; i++) {
            ps.setObject(i + 1, args[i]);
        }
        ps.execute();
    } catch (SQLException e) {
        e.printStackTrace();
    } finally {
        JDBCUtils.closeResource(null, ps);
    }
}

@Test
public void testUpdateWithTx() throws SQLException {
    Connection connection = null;
    try {
        connection = JDBCUtils.getConnection();
        connection.setAutoCommit(false);
        String sql1 = "update test.user_table set balance = balance - 100 where user = ?";
        updateWithTransaction(connection, sql1, "AA");
        //中间出现异常
        System.out.println(10 / 0);

        String sql2 = "update test.user_table set balance = balance + 100 where user = ?";
        updateWithTransaction(connection, sql2, "BB");
        System.out.println("提交成功");
        connection.commit();
    } catch (IOException | ClassNotFoundException | SQLException e) {
        //回滚数据
        assert connection != null;
        connection.rollback();
        e.printStackTrace();
    } finally {
        assert connection != null;
        connection.setAutoCommit(true);//将自动提交改回默认状态,在数据库连接池中可能会影响别人
        JDBCUtils.closeResource(connection, null);
    }
}

在MySQL中可以设置隔离级别,应对脏读,不可重复读,幻读问题:

set session transaction isolation level serializable;#设置当前会话的隔离级别

set global transaction isolation level read committed;#设置全局隔离级别

在Java中也可以设置:

int transactionIsolation = connection.getTransactionIsolation();//获取隔离级别

int TRANSACTION_NONE = 0;//不支持事务
int TRANSACTION_READ_UNCOMMITTED = 1;//允许读未提交数据
int TRANSACTION_READ_COMMITTED = 2;//允许事务读取已经提交的变更,可避免脏读
int TRANSACTION_REPEATABLE_READ = 4;//REPEATABLE READ:确保事务多次从同一个字段读取出相同的值,在这个事务期间禁止其他事务修改该字段,可以避免脏读和不可重复读(默认级别)
int TRANSACTION_SERIALIZABLE = 8;//类似于Java中的加锁,可以避免所有并发问题,但性能较低

connection.setTransactionIsolation(Connection.TRANSACTION_SERIALIZABLE);//设置隔离级别

对于数据库操作,一般是以下逻辑:

xxxBean.java//定义数据对象
BaseDAO.java//定义基本的通用的增删改查数据库操作
xxxDAO.java//该接口针对特定的bean定义数据库操作规范
xxxDaoImpl.java//实现xxxDAO接口,并继承BaseDAO,利用通用的增删改查操作实现对特定对象的数据库操作

BaseDAO中提供了增删改查等操作的关键代码,但是真正进行增删改查操作的是在xxxDaoImpl中,在这里传入具体的SQL语句给继承来的insert,delete等方法.

        String sql = "insert into test.customers(name,email,birth) values(?,?,?)";
        int count = runner.update(connection, sql, "蔡徐坤", "cxk@gmail.com", "1997-08-14");

    } catch (SQLException e) {

@Test//自定义ResultSetHandler实现类
public void QueryTest4() throws SQLException {
    Connection connection = null;
    try {
        QueryRunner runner = new QueryRunner();
        connection = JDBCUtils.getConnection3();
        String sql = "select id,name,email,birth from test.customers where id = ?";
        ResultSetHandler<Customer> handler = new ResultSetHandler<>() {
            @Override
            public Customer handle(ResultSet resultSet) throws SQLException {//传入了结果集
                if (resultSet.next()) {
                    int id = resultSet.getInt("id");
                    String name = resultSet.getString("name");
                    String email = resultSet.getString("email");
                    java.sql.Date birth = resultSet.getDate("birth");
                    return new Customer(id, name, email, birth);
                }
                return null;
            }
        };
        Customer customer = runner.query(connection, sql, handler, 23);

    } catch (SQLException e) {
        e.printStackTrace();
    } finally {
        JDBCUtils.closeResource(connection, null);
    }
}
