# 基础查询

 语法：

 */

#查询单个字段
select last_name
from employees;
#查多个字段
select last_name, salary, email
from employees;
#查所有字段
select *
from employees;
#当字段名和关键字重复时，给字段加上''

#查询常量值
select 100;
select 'aa';

#查询表达式
select 5 * 8;

select 100 % 98 as 结果;

#加号
/*
 MySQL 中的加号只有运算功能，若是有字符，则会试着将字符转为数值
 ‘123’转换成功，则继续加法
 ‘aa’转换失败，则将字符转换成0
 如果有一方是null，则结果是null
 */
#将员工名和姓拼在一起，显示为姓名
select concat(last_name, first_name) as 姓名
from employees;

/*
 将奖金率字段和其它字段拼接，因为奖金率字段有部分为null,导致部分结果为null
 可以使用ifnull函数过滤
 */
select ifnull(commission_pct, 0) as 奖金率
from employees;#null的地方换成0

select concat(last_name, ',', first_name, ',', ifnull(commission_pct, 0)) as result
from employees;

第二部分：

#条件查询
/*
 select③
    查询列表
 from①
    表名
 where②
    筛选条件

 分类：
 一、按条件表达式筛选：条件运算符 ><=,不等于是<>，!=;
 二、按逻辑表达式筛选：&& || !或者and or not;
 三、模糊查询：like,between and,in,is null;
 */

# 按条件表达式筛选
select *
from employees
where salary > 12000;

select last_name, department_id
from employees
where department_id <> 90;

#按逻辑表达式筛选
select *
from employees
where salary <= 12000
  and salary >= 5000;

#模糊查询
#和通配符搭配使用
select *
from employees
where last_name like '%a%';#%表示任意多个字符，包括0个

select last_name, salary
from employees
where last_name like '__e_a%';#_表示一个字符

select last_name, salary
from employees
where last_name like '_\_%';#\表示转义

select last_name, salary
from employees
where last_name like '_$_%' escape '$';#用escape指定转义符号

select *
from employees
where employee_id between 100 and 120;#包含临界值,不能颠倒

select last_name, job_id
from employees
#in列表不能用通配符,因为in等价于=,而不是like
where job_id in ('IT_PROG', 'AD_VP', 'AD_PRES');#判断字段中的值是否是某一项,in列表的值必须一致或兼容

select last_name, commission_pct
from employees
where commission_pct is null;
#不能用等于号来判断null

#安全等于
select last_name, commission_pct
from employees
where commission_pct <=> null;

select last_name, salary
from employees
where salary <=> 12000;

#注意点
select *
from employees where commission_pct is not null ;

select *
from employees
where commission_pct like '%';#通配符不能表示null
/*
 排序查询
 select 查询列表 from 表 [where 筛选条件] order by 排序条件 [asc|desc]
 order by子句一般放在最后面,limit子句除外,排序条件可以多个字段,表达式,函数,别名

select *

/*
 case语句:
 case 要判断的表达式
 when 常量值 then 语句或者值;(当是值时,不加分号)
 ...
 else 默认语句
 end
 */
select salary  as 原始工资,
       department_id,
       case department_id
           when 30 then salary * 1.1
           when 40 then salary * 1.2
           when 50 then salary * 1.3
           else salary
           end as 新工资
from employees;
/*
 case还可以做多重if判断
 case
 when 条件 then 语句或者值;(当是值时,不加分号)
 ...
 else 默认语句
 end
 */
select salary,
       case
           when salary > 20000 then 'A'
           when salary > 15000 then 'B'
           when salary > 10000 then 'C'
           else 'D'
           end as 工资级别
from employees;

#一,SQL92标准
#1,等值连接
/*
 多表等值连接的结果为多表的交集部分
  n表连接至少要n-1个连接条件
  多表的顺序没有要求
 */
select name, boyName
from boys,
     beauty
where beauty.boyfriend_id = boys.id;

use myemployees;
select last_name, department_name
from employees,
     departments
where employees.department_id = departments.department_id;

select last_name, e.job_id, job_title
from employees as e,
     jobs as j
where e.job_id = j.job_id;

select last_name, department_name
from employees,
     departments
where employees.department_id = departments.department_id
  and employees.commission_pct is not null;#添加筛选

select department_name, city
from departments d,
     locations l
where d.location_id = l.location_id
  and city like '_o%';

select city, count(*) as 个数
from locations l,
     departments d
where d.location_id = l.location_id
group by city;

select department_name, d.manager_id, min(salary)
from departments d,
     employees e
where d.department_id = e.department_id
  and commission_pct is not null
group by department_name, d.manager_id;

select job_title, count(*)
from employees e,
     jobs j
where j.job_id = e.job_id
group by job_title
order by count(*) desc;

select last_name, department_name, city
from employees e,
     departments d,
     locations l
where e.department_id = d.department_id
  and d.location_id = l.location_id;

#2,非等值连接
#查询员工的工资和工资等级
select salary, grade_level
from employees e,
     job_grades g

#3,自连接(查员工的领导名字)

#二,SQL99标准
/*
 select 查询列表
 from 表一 别名 [连接类型]
 join 表二 别名
 on 连接条件
 [where 筛选条件]
 内连接:inner
 外连接;
    左外:left [outer]
    右外:right [outer]
    全外:full [outer]
 交叉连接:cross
 */

#1,内连接
/*
 内连接又分为等值连接,非等值连接,自连接,内连接是查表的交集
 */
#等值连接
select last_name, department_name
from employees e
         inner join departments d on e.department_id = d.department_id;

select last_name, job_title
from employees e
         inner join jobs j on e.job_id = j.job_id
where last_name like '%e%';

select city, count(*)
from departments d
         inner join locations l on d.location_id = l.location_id
group by city
having count(*) > 3;

select count(*), department_name
from departments
         inner join employees e on departments.department_id = e.department_id
group by department_name
having count(*) > 3
order by count(*) desc;

select last_name, department_name, job_title
from employees e
         inner join departments d on e.department_id = d.department_id
         inner join jobs j on e.job_id = j.job_id
order by department_name desc;

#非等值连接
select salary, grade_level
from employees e
         inner join job_grades g
                    on e.salary between g.lowest_sal and highest_sal;

#自连接
select e.last_name, m.last_name
from employees e
         inner join employees m
                    on e.manager_id = m.employee_id
where e.last_name like '%k%';

#二,外连接,查一个表有,另一个表没有的数据
#外连接查询的结果为主表中的所有记录,如果没有匹配的,就像是null
#外连接结果=内连接+主表有从表没有的记录
#左外连接,left join左边的是主表
#右外连接,right join右边的是主表
#左外和右外交换两表顺序效果一样

#查询没有男朋友的女生
use girls;
select b.name, bo.*
from beauty b
         left join boys bo
                   on boyfriend_id = bo.id
where bo.id is null;

select b.*, bo.*
from boys bo
         left join beauty b
                   on b.boyfriend_id = bo.id
where bo.id is null;

use myemployees;
#查询哪个部门没有员工
select d.*, e.employee_id
from departments d
         left join employees e on d.department_id = e.department_id
where e.employee_id is null;

#交叉连接
select b.*, bo.*
from beauty b
         cross join boys bo;

select b.id, b.name, bo.*
from beauty b
         left join boys bo
                   on b.boyfriend_id = bo.id
where b.id > 3;

use myemployees;
select city, departments.*

         left join departments
                   on locations.location_id = departments.location_id
where departments.location_id is null;
#子查询
/*
 出现在其它语句中的select语句,称为子查询或内查询
 外部的查询称为主查询

 分类:
    select后面:仅支持标量子查询
    from后面:表子查询
    where和having后面:标量子查询⭐,列子查询⭐,行子查询
    exists后面(相关子查询):表子查询
 按结果集的行列数:
    标量子查询(结果集只有一行一列)
    列子查询(结果集只有一列多行)
    行子查询(结果集只有一行多列)
    表子查询(结果集一般为多行多列)
 */

#一,where和having后面
/*
 子查询放在小括号内
 子查询一般放在条件右侧
 标量子查询一般搭配单行操作符使用
 列子查询一般搭配多行操作 in,all,any/some
 */
#标量子查询
#查谁的工资表Abel高
select *
from employees
where salary > (select salary
                from employees
                where last_name = 'Abel');

#查询job_id与141号员工相同,salary比143号员工多的员工
select last_name, job_id, salary
from employees
where job_id = (select job_id from employees where employee_id = 141)
  and salary > (select salary from employees where employee_id = 143);

#列子查询(多行子查询)
#查location_id是14000或1700的部门中的所有员工姓名
select last_name
from employees
where department_id in (
    select distinct department_id
    from departments
    where location_id in (1400, 1700)
);

#行子查询
#查询员工编号最小并且工资最高的员工信息
#写法一
select *
from employees
where employee_id = (select min(employee_id) from employees)
  and salary = (select max(salary) from employees);
#写法二
select *
from employees
where (employee_id, salary) = (select min(employee_id), max(salary) from employees);

#二,select后面
#查询每个部门的员工个数
select departments.*, (select count(*) from employees e where e.department_id = departments.department_id)
from departments;

select d.*, count(*) as 个数
from departments d
         inner join employees e on d.department_id = e.department_id

#查询员工号120的部门名
select (select department_name
        from departments d

#三,from后面
#查询每个部门的平均工资的工资等级
select avg(salary), department_id

group by department_id;

select ag_dep.*, grade_level
from (select avg(salary) ag, department_id
      from employees
      group by department_id) ag_dep
         inner join job_grades g
                    on ag_dep.ag between lowest_sal and highest_sal;

select *

#union联合查询
#将多条查询语句的结果合并成一个

#查询部门编号>90或邮箱包含a的员工信息
select *
from employees
where email like '%a%'
   or department_id > 90;

select *
from employees
where email like '%a%'
union
select *
from employees
where department_id > 90;
/*
 要求联合查询的多条语句的查询列数是一致的
 另外,语句间查询的字段要一直,类型,顺序
 union默认会去重,不想去重就在union后加一个all
 */

DML语言

#一,插入语言
/*
 insert into 表名(列名...) values(值...)
 */
insert into beauty(id, name, sex, borndate, phone, photo, boyfriend_id)
values (13, '唐艺昕', '女', '1990-4-23', '1895200001', null, 2);
#上面的列名和下面的值对应,也可以省略列名,但是值的顺序要和表对应

insert into beauty
set id=19,
    name='刘涛',
    phone='1596458623';

/*
 第一种可以一次插入多行,中间用逗号隔开
 第一种还可以加入子查询语句,将select结果作为值插入表
 */
insert into beauty(id, name, phone)
select 26, '宋茜', '11582222222';

#二,修改语句
/*
 update 表名
 set 列=新值,列=新值...
 where 筛选条件
 */
update beauty
set phone='12345678911'
where name like '唐%';

#修改多表
/*
 update 表一 别名
 inner|left|right join 表二 别名
 on 连接条件
 set 列=值...
 where 筛选条件;
 */

#修改没有男朋友的女神的男朋友都为2号
use girls;
update boys bo
    right join beauty b on bo.id = b.boyfriend_id
set b.boyfriend_id=2
where b.id is null;

#,三,删除语句
/*
 delete
 delete from 表名 where 条件
 */
#单表删除
delete
from beauty
where phone like '%9';
#多表删除
/*
 delete 表一别名,表二别名
 from 表一 别名
 inner|left|right join 表二 别名 on 连接条件
 where 筛选条件
 */

delete b
from beauty b
         inner join boys bo
                    on b.boyfriend_id = bo.id
where bo.boyName = '张无忌';
