ajax是用来异步的局部更新页面的技术,具体代码:

<script type="text/javascript">
    // 在这里使用javaScript语言发起Ajax请求，访问服务器AjaxServlet中javaScriptAjax
    function ajaxRequest() {
// 				1、我们首先要创建XMLHttpRequest 
        var xmlhttprequest = new XMLHttpRequest();
        //调用open方法设置请求参数,?表示请求的参数,即action;true表示异步的.
        xmlhttprequest.open("GET", "http://localhost:8080/Books/ajaxServlet?action=javaScriptAjax", true);

        //当readySatate属性改变时,就会调用onreadystatechange函数,readyState有五种状态,从0-4,4表示请求完成且响应就绪
        //这里意思就是一旦readyState状态改变就检查是不是服务器响应过来了
        xmlhttprequest.onreadystatechange = function () {
            if (xmlhttprequest.readyState == 4 && xmlhttprequest.status == 200) {
                alert(xmlhttprequest.responseText);
            }
        }
// 				3、调用send方法发送请求
        xmlhttprequest.send();


        // alert("我是最后一行的代码");

    }
</script>

一般用Ajax是用JQuery来实现:

$.ajax方法:url:请求的地址,type:请求的类型,data:发送给服务器的数据,dataType:响应的数据类型,success:请求成功,响应的回调函数,类似与上面readyState改变且符合4&200要求时所执行的代码.

其中data有两种格式:{key:value}或者name=value&name=value.

<script type="text/javascript">
$(function(){
    // ajax请求
    $("#ajaxBtn").click(function(){
        $.ajax({
            url:"http://localhost:8080/Books/ajaxServlet",
            // data:"action=jQueryAjax",
            data:{action:"jQueryAjax"},//data是发送给服务器的数据,这里指的是请求的参数
            type:"GET",
            //data表示返回的数据类型
            success:function (data) {
                // alert("服务器返回的数据是：" + data);
                // var jsonObj = JSON.parse(data);
                $("#msg").html(" ajax 编号：" + data.id + " , 姓名：" + data.name);
            },
            dataType : "json"//常用的text,json
        });
    });
}
</script>

$.get方法和$.post方法是在上面的方法基础上进一步封装,简化代码.参数有url:请求的地址,data:发送的数据,callback:成功的回调函数,type:返回的数据类型.

<script type="text/javascript">
$(function(){
    // ajax--get请求
    $("#getBtn").click(function () {
        //依次写url,data,callback,datatype,不需要写明
        $.get("http://localhost:8080/Books/ajaxServlet", "action=jQueryGet", function (data) {
            $("#msg").html(" get 编号：" + data.id + " , 姓名：" + data.name);
        }, "json");
    });

    // ajax--post请求
    $("#postBtn").click(function () {
        // post请求
        $.post("http://localhost:8080/Books/ajaxServlet", "action=jQueryPost", function (data) {
            $("#msg").html(" post 编号：" + data.id + " , 姓名：" + data.name);
        }, "json");
    });
}
</script>

serialize()方法:表单中所有表单项的内容全部获取到,并以name=value&name=value的形式拼接.

$("#submit").click(function () {
    // 把参数序列化,在data中,首先把请求参数action写明,然后用&将表单中的参数连接起来一起提交
    $.getJSON("http://localhost:8080/16_json_ajax_i18n/ajaxServlet", "action=jQuerySerialize&" + $("#form01").serialize(), function (data) {
        $("#msg").html(" Serialize 编号：" + data.id + " , 姓名：" + data.name);
    });
});

用Ajax实现校验用户名是否存在:

$("#username").blur(function () {
    //1 获取用户名
    var username = this.value;
    $.getJSON("http://localhost:8080/book/userServlet","action=ajaxExistsUsername&username=" + username,function (data) {
        // existsUsername是返回的json中的key名
        if (data.existsUsername) {
            $("span.errorMsg").text("用户名已存在！");
        } else {
            $("span.errorMsg").text("用户名可用！");
        }
    });
});
