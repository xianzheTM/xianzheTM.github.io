下面是反射的用处的一些展示：

package com.ylxianzhe.Reflect;

import java.lang.reflect.Constructor;
import java.lang.reflect.Field;
import java.lang.reflect.Method;

public class Test {
    @org.junit.Test
    public void test() {
        //正常情况下，私有属性和方法无法使用
        Person person = new Person("Tom", 12);
        person.age = 10;
        System.out.println(person.toString());
        person.showPerson();
    }

    @org.junit.Test
    public void test2() throws Exception {
        Class<Person> personClass = Person.class;
        //通过反射创建Person类的对象
        Constructor<Person> con = personClass.getConstructor(String.class, int.class);
        Person tom = con.newInstance("Tom", 13);
        //用反射调属性
        Field age = personClass.getDeclaredField("age");
        age.set(tom, 11);
        //用反射调方法
        Method show = personClass.getDeclaredMethod("showPerson");
        show.invoke(tom);

        //用反射来访问私有构造器
        Constructor<Person> con1 = personClass.getDeclaredConstructor(String.class);
        con1.setAccessible(true);
        Person jerry = con1.newInstance("Jerry");
        //用反射修改私有属性
        Field name = personClass.getDeclaredField("name");
        name.setAccessible(true);
        name.set(jerry, "Buluto");
        System.out.println(jerry);
        //调私有方法
        Method showNation = personClass.getDeclaredMethod("showNation", String.class);
        showNation.setAccessible(true);
        String nation = (String) showNation.invoke(jerry, "china");
    }
}

当使用Java命令将字节码文件加载到内存后，内存中的这个类就称为运行时类。一个Class的实例就对应这一个运行时类。获取Class的实例的方式有四种：

//通过不同方法获得的Class实例其实是同一个，加载到内存中的运行时类会缓存一段时间
//方式一
Class personclass1 = Person.class;
//方式二
Person p = new Person();
Class personclass2 = p.getClass();
//方式三
Class personclass3 = Class.forname("com.study.Reflect.Person")//类名写完整
//方式四，使用类的加载器：ClassLoader
ClassLoader classloader = Test.class.getClassLoader();
Class personclass4 = classloader.loadClass("com.study.Reflect.Person");

下面两个测试获取了Person的运行时类各种属性的一些信息：

public class Test2 {
    @Test
    public void test1() {
        Class<Person> personClass = Person.class;
        Field[] fields = personClass.getFields();//获取运行时类和其父类中public的属性
        for (Field f : fields) {
            System.out.println(f);
            //public int com.ylxianzhe.Reflect.Person.id
            //public double com.ylxianzhe.Reflect.Creature.weight
        }

        Field[] declaredFields = personClass.getDeclaredFields();//获取运行时类的所有属性，不包含父类的
        for (Field declaredField : declaredFields) {
            System.out.println(declaredField);
            //private java.lang.String com.ylxianzhe.Reflect.Person.name
            //int com.ylxianzhe.Reflect.Person.age
            //public int com.ylxianzhe.Reflect.Person.id
        }
    }

    @Test
    public void test2() {
        Class<Person> personClass = Person.class;
        Field[] declaredFields = personClass.getDeclaredFields();
        for (Field f : declaredFields) {
            //获取权限修饰符
            System.out.print(Modifier.toString(f.getModifiers())+'\t');
            //获取数据类型
            Class<?> type = f.getType();
            System.out.print(type.getName()+'\t');
            //获取变量名
            System.out.println(f.getName());
        }
    }
}

下面是获取指定结构的方法：

@Test
public void test6() throws  Exception{
    Class<Person> personClass = Person.class;
    //创建运行时类的对象,新方式，上面的过时了
    Person person = personClass.getDeclaredConstructor().newInstance();
    //获取指定的属性，要求是public的
    Field id = personClass.getField("id");
    //设置属性值,set方法参数一指定设置的对象，参数二指定设置值
    id.set(person, 123);
    //获得指定对象的属性值
    int pId = (int) id.get(person);

    //以上方法不常用
    //用getDeclaredField来获取运行时类的所有方法，包括私有的
    Field name = personClass.getDeclaredField("name");
    //要想使用私有的还需设置权限
    name.setAccessible(true);
    name.set(person,"Tom");
    System.out.println(name.get(person));

    //获取指定的方法,第一个参数指定方法名，后面的可变形参指定方法的参数列表
    Method show = personClass.getDeclaredMethod("show", String.class);
    show.setAccessible(true);
    //invoke第一个参数是方法的调用者，后面的可变形参指定实参列表
    //invoke方法的返回值就是对应类中调用方法的返回值
    String chn = (String) show.invoke(person, "CHN");

    //调用静态方法
    Method showDesc = personClass.getDeclaredMethod("showDesc");
    showDesc.setAccessible(true);
    //静态方法的调用者是类，也可以写个null，因为showDesc本身就是通过personClass这个运行时类来的，自然知道静态方法的调用者
    Object o = showDesc.invoke(Person.class);
    System.out.println(o);//没有返回值就是null
}

学会反射后就用来做动态代理，下面是静态代理：

package com.ylxianzhe.Reflect.Proxy;

interface ClothFactory {
    void produceCloth();
}

//代理类
class ProxyClothFactory implements ClothFactory {
    private ClothFactory factory;//用被代理类对象实例化

    public ProxyClothFactory(ClothFactory factory) {
        this.factory = factory;
    }

    @Override
    public void produceCloth() {
        System.out.println("代理工厂做一些准备工作");
        factory.produceCloth();
        System.out.println("代理工厂做一些后续工作");
    }
}

//被代理类
class NikeClothFactory implements ClothFactory {
    @Override
    public void produceCloth() {
        System.out.println("Nike生产运动服");
    }
}
public class StaticProxy {
    public static void main(String[] args) {
        //创建被代理类对象
        NikeClothFactory nike = new NikeClothFactory();
        //创建代理类对象
        ClothFactory proxyClothFactory = new ProxyClothFactory(nike);
        proxyClothFactory.produceCloth();
    }
}

首先要有一个代理类和被代理类都要实现的接口，接着代理类内部包含一个被代理类，实际执行方法用的就是这个内部的被代理类。

package com.ylxianzhe.Reflect.Proxy;

import java.lang.reflect.InvocationHandler;
import java.lang.reflect.Method;
import java.lang.reflect.Proxy;

interface Human {
    String getBelief();

    void eat(String food);
}

//被代理类
class SuperMan implements Human {
    @Override
    public String getBelief() {
        return "I can fly";
    }

    @Override
    public void eat(String food) {
        System.out.println("我喜欢吃" + food);
    }
}

//根据加载的被代理类动态的创建一个代理类及其对象
//通过代理类的对象动态的调用被代理类的同名方法
class ProxyFactory {
    //调用此方法，返回一个代理类的对象，object：被代理类对象
    public static Object getProxyInstance(Object object) {
        MyInvocationHandler handler = new MyInvocationHandler();
        handler.bind(object);
        //第二个参数得到被代理类实现的接口，第三个参数得到代理类需要做的方法
        return Proxy.newProxyInstance(object.getClass().getClassLoader(), object.getClass().getInterfaces(), handler);
    }
}

class MyInvocationHandler implements InvocationHandler {//Handler是代理类中对被代理对象的处理方法
    private Object object;//需要使用被代理类的对象进行赋值，由于真正的方法运行还是需要被代理类，所以代理类要包含一个被代理类

    public void bind(Object obj) {//getProxyInstance传入的被代理类被Handler包含
        this.object = obj;
    }

    //当通过代理类的对象调用方法a时，就会自动调用下面的方法
    //将被代理类要执行的方法a的功能放在invoke方法内
    @Override
    public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
        //method即为代理类对象调用的方法，此方法也就作为被代理类对象调用的方法
        //object:被代理类对象
        //method的返回值就作为*当前类*的invoke方法的返回值
        return method.invoke(object, args);//利用反射，执行object的method方法
        //Handler还可以在此处做一些增强方法
    }
}

public class ProxyTest {
    public static void main(String[] args) {
        SuperMan superMan = new SuperMan();
        Human proxyInstance = (Human) ProxyFactory.getProxyInstance(superMan);
        proxyInstance.eat("火锅");
        System.out.println(proxyInstance.getBelief());

        NikeClothFactory nikeClothFactory = new NikeClothFactory();
        ClothFactory proxyClothFactory = (ClothFactory) ProxyFactory.getProxyInstance(nikeClothFactory);
        proxyClothFactory.produceCloth();
    }
}
